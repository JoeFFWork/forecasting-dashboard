function loadTimeline() { var store = document.getElementById('timelineStore').value; var commodity = document.getElementById('timelineCommodity').value; var query = (store ? '&store=' + store : '') + (commodity ? '&commodity=' + commodity : ''); fetch(API_URL + '/api/forecast-timeline?weeks=12' + query) .then(function(res) { return res.json(); }) .then(function(timeline) { var content = document.getElementById('timelineContent'); content.innerHTML = ''; if (!timeline || timeline.length === 0) { content.innerHTML = '

No timeline data available

'; return; } // Get list of commodities that have non-zero data across all 12 weeks var storesWithData = {}; timeline.forEach(function(week) { Object.keys(week.stores).forEach(function(storeKey) { if (!storesWithData[storeKey]) { storesWithData[storeKey] = {}; } Object.keys(week.stores[storeKey]).forEach(function(comm) { var data = week.stores[storeKey][comm]; if (!storesWithData[storeKey][comm]) { storesWithData[storeKey][comm] = { hasStock: false, hasTotals: false }; } // Check if this week has any real data if (data.startStock > 0 || data.incoming > 0 || data.endStock > 0) { storesWithData[storeKey][comm].hasStock = true; } if (data.startStock > 0 || data.collections > 0 || data.incoming > 0 || data.endStock > 0) { storesWithData[storeKey][comm].hasTotals = true; } }); }); }); // Group by store var stores = store ? [store] : ['Newport', 'Kenyon', 'Avonmouth', 'Halder', 'Garston']; var hasAnyData = false; stores.forEach(function(storeKey) { var commoditiesToShow = []; // Only include commodities with actual stock data if (storesWithData[storeKey]) { Object.keys(storesWithData[storeKey]).forEach(function(comm) { if (storesWithData[storeKey][comm].hasStock) { commoditiesToShow.push(comm); } }); } if (commodity) { if (commoditiesToShow.indexOf(commodity) !== -1) { commoditiesToShow = [commodity]; } else { commoditiesToShow = []; } } if (commoditiesToShow.length === 0) return; hasAnyData = true; var storeTimeline = document.createElement('div'); storeTimeline.className = 'store-timeline'; storeTimeline.innerHTML = '
📍 ' + storeKey + '
'; commoditiesToShow.sort().forEach(function(comm) { var row = document.createElement('div'); row.className = 'commodity-row'; var label = document.createElement('div'); label.className = 'commodity-label'; label.textContent = comm; row.appendChild(label); for (var w = 0; w < 12; w++) { var weekData = timeline[w]; var cell = document.createElement('div'); cell.className = 'week-cell'; if (!weekData || !weekData.stores || !weekData.stores[storeKey] || !weekData.stores[storeKey][comm]) { cell.textContent = '—'; } else { var data = weekData.stores[storeKey][comm]; if (data.stockoutRisk) { cell.classList.add('danger'); } else if (data.utilizationPercent > 80) { cell.classList.add('warning'); } else { cell.classList.add('good'); } cell.innerHTML = '
' + data.endStock.toFixed(0) + ' MT
' + '
↓' + data.collections.toFixed(0) + '
' + '
↑' + data.incoming.toFixed(0) + '
'; } row.appendChild(cell); } storeTimeline.appendChild(row); }); content.appendChild(storeTimeline); }); if (!hasAnyData) { content.innerHTML = '

No data available for selected filters

'; } }); }
