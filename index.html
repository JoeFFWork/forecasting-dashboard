async function loadTimeline() { const store = document.getElementById('timelineStore').value; const commodity = document.getElementById('timelineCommodity').value; try { const query = `${store ? `&store=${store}` : ''}${commodity ? `&commodity=${commodity}` : ''}`; const res = await fetch(`${API_URL}/api/forecast-timeline?weeks=12${query}`); const timeline = await res.json(); const content = document.getElementById('timelineContent'); content.innerHTML = ''; if (!timeline || timeline.length === 0) { content.innerHTML = '

No timeline data available

'; return; } // Get list of commodities that actually have data const storesWithData = {}; timeline.forEach(week => { Object.keys(week.stores).forEach(storeKey => { if (!storesWithData[storeKey]) { storesWithData[storeKey] = new Set(); } Object.keys(week.stores[storeKey]).forEach(comm => { storesWithData[storeKey].add(comm); }); }); }); // Group by store const stores = store ? [store] : ['Newport', 'Kenyon', 'Avonmouth', 'Halder', 'Garston']; stores.forEach(storeKey => { const storeTimeline = document.createElement('div'); storeTimeline.className = 'store-timeline'; storeTimeline.innerHTML += `
📍 ${storeKey}
`; // Only show commodities that have data in this store const commoditiesToShow = storesWithData[storeKey] ? Array.from(storesWithData[storeKey]) : []; if (commodity) { // If specific commodity selected, only show if it has data if (commoditiesToShow.includes(commodity)) { commoditiesToShow.length = 0; commoditiesToShow.push(commodity); } else { commoditiesToShow.length = 0; } } commoditiesToShow.sort().forEach(comm => { const row = document.createElement('div'); row.className = 'commodity-row'; const label = document.createElement('div'); label.className = 'commodity-label'; label.textContent = comm; row.appendChild(label); for (let w = 0; w < 12; w++) { const weekData = timeline[w]; if (!weekData || !weekData.stores || !weekData.stores[storeKey] || !weekData.stores[storeKey][comm]) { const cell = document.createElement('div'); cell.className = 'week-cell'; cell.textContent = '—'; row.appendChild(cell); continue; } const data = weekData.stores[storeKey][comm]; const cell = document.createElement('div'); cell.className = 'week-cell'; if (data.stockoutRisk) { cell.classList.add('danger'); } else if (data.utilizationPercent > 80) { cell.classList.add('warning'); } else { cell.classList.add('good'); } cell.innerHTML = `
${data.endStock.toFixed(0)} MT
↓${data.collections.toFixed(0)}
↑${data.incoming.toFixed(0)}
`; row.appendChild(cell); } storeTimeline.appendChild(row); }); if (commoditiesToShow.length > 0) { content.appendChild(storeTimeline); } }); if (content.innerHTML === '') { content.innerHTML = '

No data available for selected filters

'; } } catch (error) { console.error('Error:', error); } }
